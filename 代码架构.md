# 信息使能AI平台 — 重构后的 Skills 代码架构

---

## 一、整体代码结构设计原则

```
设计原则：
1. 单一职责：每个 Skill 只做一件事
2. 依赖注入：组件间通过接口依赖，不硬编码
3. 基类抽象：同类 Skill 继承同一基类，消除重复逻辑
4. Pipeline 组合：多个 Skill 通过 Pipeline 串联，不互相耦合
5. 统一输入输出：所有 Skill 使用标准 Document / Result 对象
```

```python
from dataclasses import dataclass, field
from enum import Enum
from typing import Any

class Severity(str, Enum):
    ERROR   = "error"
    WARNING = "warning"
    INFO    = "info"

class SkillType(str, Enum):
    INGESTION  = "ingestion"
    VALIDATION = "validation"
    COMPARISON = "comparison"
    AI         = "ai"
    RAG        = "rag"

@dataclass
class Metadata:
    doc_id:             str  = ""
    title:              str  = ""
    version:            str  = ""
    product_line:       str  = ""
    module:             str  = ""
    department:         str  = ""
    doc_type:           str  = ""
    confidential_level: str  = "internal"
    owner:              str  = ""
    effective_date:     str  = ""
    source_path:        str  = ""

@dataclass
class Chunk:
    chunk_id:   str
    doc_id:     str
    content:    str
    chunk_type: str          # paragraph | table | step | title
    metadata:   Metadata = field(default_factory=Metadata)
    embedding:  list[float] = field(default_factory=list)

@dataclass
class Document:
    doc_id:   str
    raw_path: str
    content:  str            # 原始文本（解析后）
    chunks:   list[Chunk]    = field(default_factory=list)
    metadata: Metadata       = field(default_factory=Metadata)

@dataclass
class Issue:
    rule_id:   str
    severity:  Severity
    message:   str
    location:  str           # xpath / line number / chunk_id
    suggestion: str = ""

@dataclass
class SkillResult:
    skill_id: str
    doc_id:   str
    passed:   bool
    issues:   list[Issue]        = field(default_factory=list)
```

### `core/base.py` — 所有 Skill 的抽象基类

```python
# core/base.py
from abc import ABC, abstractmethod
from .models import Document, SkillResult, SkillType
import logging

logger = logging.getLogger(__name__)

class BaseSkill(ABC):
    """所有 Skill 的统一基类，负责日志、异常捕获、结果封装"""

    skill_id:   str      = "base"
    skill_type: SkillType = SkillType.VALIDATION

    def run(self, document: Document) -> SkillResult:
        logger.info("[%s] 开始处理: %s", self.skill_id, document.doc_id)
        try:
            result = self._execute(document)
        except Exception as exc:
            logger.exception("[%s] 执行异常", self.skill_id)
            result = SkillResult(
                skill_id=self.skill_id,
                doc_id=document.doc_id,
                passed=False,
                errors=[str(exc)],
            )
        logger.info("[%s] 完成，passed=%s", self.skill_id, result.passed)
        return result

    @abstractmethod
    def _execute(self, document: Document) -> SkillResult: ...
```

### `core/pipeline.py` — Pipeline 编排器

```python
# core/pipeline.py
from dataclasses import dataclass, field
from .base import BaseSkill
from .models import Document, SkillResult

@dataclass
class PipelineResult:
    doc_id:       str
    results:      list[SkillResult] = field(default_factory=list)
    all_passed:   bool = True

    # 便捷聚合：收集所有 issues
    def all_issues(self):
        return [issue for r in self.results for issue in r.issues]

    # 便捷聚合：找到某个 skill 的结果
    def get(self, skill_id: str) -> SkillResult | None:
        return next((r for r in self.results if r.skill_id == skill_id), None)


class Pipeline:
    """
    有序 Skill 链，支持：
    - 顺序执行
    - 条件跳过（skip_on_error）
    - 结果透传（前一个 skill 的 data 可传入下一个）
    """

    def __init__(self, skills: list[BaseSkill], skip_on_error: bool = False):
        self._skills       = skills
        self._skip_on_error = skip_on_error

    def run(self, document: Document) -> PipelineResult:
        pipeline_result = PipelineResult(doc_id=document.doc_id)

        for skill in self._skills:
            result = skill.run(document)
            pipeline_result.results.append(result)

            if not result.passed:
                pipeline_result.all_passed = False
                if self._skip_on_error:
                    break

            # 将 skill 输出透传到 document.extra，供后续 skill 读取
            document.extra[skill.skill_id] = result.data

        return pipeline_result
```

### `core/registry.py` — Skill 注册中心（消除硬编码）

```python
# core/registry.py
from .base import BaseSkill

class SkillRegistry:
    _registry: dict[str, type[BaseSkill]] = {}

    @classmethod
    def register(cls, skill_cls: type[BaseSkill]):
        """装饰器：自动注册"""
        cls._registry[skill_cls.skill_id] = skill_cls
        return skill_cls

    @classmethod
    def get(cls, skill_id: str) -> BaseSkill:
        if skill_id not in cls._registry:
            raise KeyError(f"Skill '{skill_id}' 未注册")
        return cls._registry[skill_id]()

    @classmethod
    def list_all(cls) -> list[str]:
        return list(cls._registry.keys())
```

### `skills/validation/base_validator.py`

```python
# skills/validation/base_validator.py
from core.base import BaseSkill
from core.models import Document, SkillResult, Issue, SkillType, Severity

class BaseValidator(BaseSkill):
    """
    所有校验类 Skill 的共同基类
    子类只需实现 _validate()，返回 Issue 列表
    SkillResult 构建、passed 判断由基类统一处理
    """
    skill_type      = SkillType.VALIDATION
    fail_on_error   = True   # 有 ERROR 级别则 passed=False

    def _execute(self, document: Document) -> SkillResult:
        issues = self._validate(document)
        has_error = any(i.severity == Severity.ERROR for i in issues)
        return SkillResult(
            skill_id=self.skill_id,
            doc_id=document.doc_id,
            passed=not (self.fail_on_error and has_error),
            issues=issues,
        )

    def _validate(self, document: Document) -> list[Issue]:
        raise NotImplementedError
```

```python
# skills/validation/structure_validator.py
from lxml import etree
from core.models import Document, Issue, Severity
from core.registry import SkillRegistry
from .base_validator import BaseValidator

REQUIRED_ELEMENTS = {
    "topic":     ["title", "shortdesc", "body"],
    "task":      ["title", "shortdesc", "taskbody"],
    "concept":   ["title", "shortdesc", "conbody"],
    "reference": ["title", "refbody"],
}

@SkillRegistry.register
class StructureValidator(BaseValidator):
    skill_id = "B-01"

    def _validate(self, document: Document) -> list[Issue]:
        issues: list[Issue] = []
        try:
            root = etree.fromstring(document.content.encode())
        except etree.XMLSyntaxError as e:
            return [Issue("B-01-xml", Severity.ERROR, f"XML 解析失败: {e}", "/")]

        topic_type = root.tag
        required   = REQUIRED_ELEMENTS.get(topic_type, [])

        for element in required:
            if root.find(f".//{element}") is None:
                issues.append(Issue(
                    rule_id    = f"B-01-{element}",
                    severity   = Severity.ERROR,
                    message    = f"<{topic_type}> 缺少必填元素 <{element}>",
                    location   = f"/{topic_type}",
                    suggestion = f"请补充 <{element}> 元素",
                ))
        return issues
```

### `skills/validation/reference_validator.py`

```python
# skills/validation/reference_validator.py
from core.models import Document, Issue, Severity
from core.registry import SkillRegistry
from .base_validator import BaseValidator

@SkillRegistry.register
class ReferenceValidator(BaseValidator):
    skill_id = "B-02"

    def _validate(self, document: Document) -> list[Issue]:
        # 从 document.extra 读取上游 A-01 解析的引用关系图（无需重复解析）
        ref_graph: dict = document.extra.get("A-01", {}).get("ref_graph", {})
        issues: list[Issue] = []

        for ref_id, target_path in ref_graph.items():
            if not self._target_exists(target_path, document):
                issues.append(Issue(
                    rule_id    = "B-02-broken-ref",
                    severity   = Severity.ERROR,
                    message    = f"引用目标不存在: {target_path}",
                    location   = ref_id,
                    suggestion = "请检查 xref/conref/keyref 的目标路径",
                ))
        return issues

    @staticmethod
    def _target_exists(path: str, document: Document) -> bool:
        known_paths: set = document.extra.get("A-05", {}).get("known_paths", set())
        return path in known_paths
```

```python
# skills/validation/report_builder.py
from core.models import SkillResult, Issue, Severity
from core.pipeline import PipelineResult
from jinja2 import Environment, FileSystemLoader
import json

class ReportBuilder:
    """
    与 Skill 解耦的报告构建器
    接收 PipelineResult，输出 HTML / JSON / CLI 文本
    """

    TEMPLATE_DIR = "templates/"

    def to_json(self, result: PipelineResult) -> str:
        return json.dumps({
            "doc_id":     result.doc_id,
            "all_passed": result.all_passed,
            "issues":     [self._issue_dict(i) for i in result.all_issues()],
        }, ensure_ascii=False, indent=2)

    def to_html(self, result: PipelineResult) -> str:
        env      = Environment(loader=FileSystemLoader(self.TEMPLATE_DIR))
        template = env.get_template("report.html.j2")
        return template.render(result=result, Severity=Severity)

    def to_cli(self, result: PipelineResult) -> str:
        lines = [f"文档: {result.doc_id}  总体: {'✅ 通过' if result.all_passed else '❌ 未通过'}"]
        for issue in result.all_issues():
            icon = {"error": "🔴", "warning": "🟡", "info": "🔵"}.get(issue.severity, "")
            lines.append(f"  {icon} [{issue.rule_id}] {issue.message}  @ {issue.location}")
            if issue.suggestion:
                lines.append(f"       建议: {issue.suggestion}")
        return "\n".join(lines)

    @staticmethod
    def _issue_dict(issue: Issue) -> dict:
        return {
            "rule_id":    issue.rule_id,
            "severity":   issue.severity,
            "message":    issue.message,
            "location":   issue.location,
            "suggestion": issue.suggestion,
        }
```

```python
# skills/ingestion/base_parser.py
from abc import abstractmethod
from core.base import BaseSkill
from core.models import Document, SkillResult, SkillType, Metadata

class BaseParser(BaseSkill):
    """
    所有解析类 Skill 的基类
    子类只需实现 _parse()，返回 (content, metadata, extra)
    Document 构建由基类统一完成
    """
    skill_type = SkillType.INGESTION

    def _execute(self, document: Document) -> SkillResult:
        content, metadata, extra = self._parse(document.raw_path)
        document.content  = content
        document.metadata = metadata
        document.extra.update(extra)
        return SkillResult(
            skill_id=self.skill_id,
            doc_id=document.doc_id,
            passed=True,
            data={"content_length": len(content)},
        )

    @abstractmethod
    def _parse(self, path: str) -> tuple[str, Metadata, dict]: ...
```

### `skills/ingestion/dita_parser.py`

```python
# skills/ingestion/dita_parser.py
from lxml import etree
from core.models import Metadata
from core.registry import SkillRegistry
from .base_parser import BaseParser

@SkillRegistry.register
class DitaParser(BaseParser):
    skill_id = "A-01"

    def _parse(self, path: str) -> tuple[str, Metadata, dict]:
        tree = etree.parse(path)
        root = tree.getroot()

        content  = etree.tostring(root, encoding="unicode")
        metadata = self._extract_metadata(root, path)
        extra    = {
            "ref_graph":  self._extract_refs(root),
            "topic_type": root.tag,
        }
        return content, metadata, extra

    @staticmethod
    def _extract_metadata(root: etree._Element, path: str) -> Metadata:
        prolog = root.find(".//prolog")
        return Metadata(
            title       = (root.findtext("title") or "").strip(),
            version     = (prolog.findtext(".//vrm/@version") if prolog else "") or "",
            source_path = path,
        )

    @staticmethod
    def _extract_refs(root: etree._Element) -> dict[str, str]:
        refs = {}
        for tag in ("xref", "conref", "keyref"):
            for el in root.iter(tag):
                ref_attr = el.get("href") or el.get("conref") or el.get("keyref", "")
                if ref_attr:
                    refs[f"{tag}:{id(el)}"] = ref_attr
        return refs
```

### `skills/ingestion/chunker.py`

```python
# skills/ingestion/chunker.py
from lxml import etree
from core.models import Document, Chunk, SkillResult, SkillType
from core.base import BaseSkill
from core.registry import SkillRegistry
import uuid

# 按语义块类型统一定义，新增类型只需扩展此字典
CHUNK_SELECTORS = {
    "title":     ".//title",
    "paragraph": ".//p",
    "step":      ".//step",
    "table":     ".//table",
    "note":      ".//note",
}

@SkillRegistry.register
class SemanticChunker(BaseSkill):
    skill_id   = "A-04"
    skill_type = SkillType.INGESTION

    def _execute(self, document: Document) -> SkillResult:
        chunks: list[Chunk] = []
        try:
            root = etree.fromstring(document.content.encode())
        except etree.XMLSyntaxError:
            return SkillResult(self.skill_id, document.doc_id, False,
                               errors=["XML 解析失败，无法分块"])

        for chunk_type, selector in CHUNK_SELECTORS.items():
            for el in root.findall(selector):
                text = "".join(el.itertext()).strip()
                if text:
                    chunks.append(Chunk(
                        chunk_id   = str(uuid.uuid4()),
                        doc_id     = document.doc_id,
                        content    = text,
                        chunk_type = chunk_type,
                        metadata   = document.metadata,
                    ))

        document.chunks = chunks
        return SkillResult(
            skill_id=self.skill_id,
            doc_id=document.doc_id,
            passed=True,
            data={"chunk_count": len(chunks)},
        )
```

### `skills/ai/base_ai_skill.py`

```python
# skills/ai/base_ai_skill.py
from abc import abstractmethod
from core.base import BaseSkill
from core.models import Document, SkillResult, SkillType
from config import settings
from openai import OpenAI

class BaseAISkill(BaseSkill):
    """
    所有 AI 类 Skill 的基类
    统一封装：模型调用、重试、token 限制、结果解析
    子类只需定义 _build_prompt() 和 _parse_response()
    """
    skill_type  = SkillType.AI
    model       = "qwen-max"       # 通过配置覆盖，不硬编码
    temperature = 0.2
    max_tokens  = 2048

    def __init__(self):
        self._client = OpenAI(
            api_key  = settings.LLM_API_KEY,
            base_url = settings.LLM_BASE_URL,
        )

    def _execute(self, document: Document) -> SkillResult:
        prompt   = self._build_prompt(document)
        response = self._call_llm(prompt)
        data     = self._parse_response(response)
        return SkillResult(
            skill_id=self.skill_id,
            doc_id=document.doc_id,
            passed=True,
            data=data,
        )

    def _call_llm(self, prompt: str) -> str:
        completion = self._client.chat.completions.create(
            model       = self.model,
            temperature = self.temperature,
            max_tokens  = self.max_tokens,
            messages    = [{"role": "user", "content": prompt}],
        )
        return completion.choices[0].message.content or ""

    @abstractmethod
    def _build_prompt(self, document: Document) -> str: ...

    @abstractmethod
    def _parse_response(self, response: str) -> dict: ...
```

```python
import json
from core.models import Document, Issue, Severity, SkillResult
from core.registry import SkillRegistry
from .base_ai_skill import BaseAISkill

SEMANTIC_CHECK_PROMPT = """\
你是一位技术文档审校专家，请检查以下文档片段，识别：
1. 表述重复
2. 语义歧义
3. 语气不规范（命令式/被动式混用）
4. 术语不一致

请以 JSON 格式返回，格式：
[{{"type": "重复|歧义|语气|术语", "location": "引用原文前20字", "message": "问题描述", "suggestion": "修改建议"}}]

文档内容：
{content}
"""

@SkillRegistry.register
class SemanticChecker(BaseAISkill):
    skill_id = "D-01"

    def _build_prompt(self, document: Document) -> str:
        # 只取前 3000 字，避免超 token
        return SEMANTIC_CHECK_PROMPT.format(content=document.content[:3000])

    def _parse_response(self, response: str) -> dict:
        try:
            items = json.loads(response)
        except json.JSONDecodeError:
            return {"raw": response, "issues": []}

        issues = [
            Issue(
                rule_id    = f"D-01-{item.get('type', 'unknown')}",
                severity   = Severity.WARNING,
                message    = item.get("message", ""),
                location   = item.get("location", ""),
                suggestion = item.get("suggestion", ""),
            )
            for item in items if isinstance(item, dict)
        ]
        return {"issues": issues}
```

```python
# pipelines/rag_pipeline.py
"""
RAG 问答 Pipeline（独立于文档检查，单独运行）
"""
from skills.rag.retriever import HybridRetriever
from skills.rag.reranker import BGEReranker
from skills.rag.answer_synthesizer import AnswerSynthesizer
from skills.rag.access_filter import AccessFilter

class RAGPipeline:
    def __init__(self, user_context: dict):
        self._filter     = AccessFilter(user_context)
        self._retriever  = HybridRetriever()
        self._reranker   = BGEReranker()
        self._synthesizer = AnswerSynthesizer()

    def query(self, question: str) -> dict:
        candidates = self._retriever.retrieve(question)
        filtered   = self._filter.apply(candidates)
        reranked   = self._reranker.rerank(question, filtered)
        answer     = self._synthesizer.synthesize(question, reranked)
        return answer
```

## 九、新增 Skill 的标准流程（防止未来代码扩散）

```
新增一个 Skill 只需 3 步：

1. 在对应 skills/<主线>/ 目录创建文件
2. 继承对应基类（BaseValidator / BaseParser / BaseAISkill）
3. 加上 @SkillRegistry.register 装饰器

无需修改 Pipeline、CLI、API 的任何代码。
```

**示例：新增编号规范检查**

```python
# skills/validation/numbering_validator.py
import re
from core.models import Document, Issue, Severity
from core.registry import SkillRegistry
from .base_validator import BaseValidator

@SkillRegistry.register
class NumberingValidator(BaseValidator):
    skill_id = "B-08"   # 新 ID，其余全部复用基类

    def _validate(self, document: Document) -> list[Issue]:
        issues = []
        for match in re.finditer(r"第[一二三四五六七八九十]+节", document.content):
            issues.append(Issue(
                rule_id    = "B-08-numbering",
                severity   = Severity.WARNING,
                message    = "使用了中文序号，建议改为阿拉伯数字编号",
                location   = f"char:{match.start()}",
                suggestion = "如：第 3 节",
            ))
        return issues
```

## 十、架构优化总结

| 优化点 | 优化前 | 优化后 |
|---|---|---|
| 日志/异常处理 | 每个 Skill 各自写 | `BaseSkill.run()` 统一处理 |
| `SkillResult` 构建 | 每个 Skill 重复构建 | `BaseValidator._execute()` 统一封装 |
| LLM 调用逻辑 | 每个 AI Skill 各自调用 | `BaseAISkill._call_llm()` 统一封装 |
| Prompt 模板 | 散落在方法体内 | 提升为模块级常量，统一维护 |
| 规则扩展 | 修改方法逻辑 | 只扩展 `STYLE_RULES` 列表 |
| Pipeline 组合 | 硬编码 Skill 实例列表 | `SkillRegistry` 按 ID 动态获取 |
| 报告输出 | 与 Skill 耦合 | `ReportBuilder` 独立，接收 `PipelineResult` |
| 新增 Skill | 需改多处 | 只加装饰器，其余零修改 |
